            function loadXMLDoc(){
                var req=new XMLHttpRequest();
                req.open('GET','http://localhost:8080/TableView/webresources/com.server.contacts',true);
                req.send();
                req.onreadystatechange = function() {
                    if (this.readyState == 4 && this.status == 200) {
                        myFunction(this);
                    }
                };
            }
            function myFunction(xml) {
                var i;
                var xmlDoc = xml.responseXML;
                var table="<tr><th>ID</th><th>Name</th></tr>";
                var x = xmlDoc.getElementsByTagName("contacts");
                for (i = 0;i<x.length;i++){ 
                    console.log(x);
                    table += "<tr><td>" + x[i].getElementsByTagName("id")[0].childNodes[0].nodeValue + "</td><td>" + x[i].getElementsByTagName("name")[0].childNodes[0].nodeValue + "</td></tr>";
                }
                document.getElementById("demo").innerHTML = table;
            }

        window.onload = loadXMLDoc;
            
        </script>

            function myFunction(xml) {
                var i;
                var xmlDoc = xml.responseXML;
                var table="<tr><th>ID</th><th>Username</th><th>Password</th><th>City</th><th>Contact No</th></tr>";
                var x = xmlDoc.getElementsByTagName("contacts");

                    x[i].getElementsByTagName("contactNo")[0].childNodes[0].nodeValue +

}
