from pynput import mouse
global fileLoc
def on_move(x, y):
    with open(fileLoc, "a+") as f:
        f.write(f"Pointer moved to {(x,y)}\n")
def on_click(x, y, button, pressed):

    with open(fileLoc, "a+") as f:

        f.write(f"{'Pressed' if pressed else 'Released'} at {(x,y)}\n")

    if not pressed:

        # Stop listener

        return False

def on_scroll(x, y, dx, dy):

    with open(fileLoc, "a+") as f:

        f.write(f"Scrolled {'down' if dy < 0 else 'up'} at {(x,y)}\n")

if __name__ == "__main__":

    fileLoc = "C:\\4812_TYCS\\temp\\process_log\\mouse_log.txt"

    # Collect events until released

    with mouse.Listener(

            on_move=on_move,

            on_click=on_click,

            on_scroll=on_scroll) as listener:

        listener.join()
