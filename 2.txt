            var request =new XMLHttpRequest();
            request.open('GET','http://localhost:8080/server/webresources/com.server.userinfo/');
            
            request.onload=function()
            {
                
                var parser = new DOMParser();
                var xmlDoc = parser.parseFromString(this.response,"text/xml");
                
                var x=xmlDoc.getElementsByTagName("userinfo");

                 //for displaying all data
                for(var i=0;i<x.length;i++){
                    console.log("he");
                   var table=document.getElementById("myTable");
                   var row=table.insertRow(-1);
                   var cell1=row.insertCell(0);
                   var cell2=row.insertCell(1);
                   cell1.innerHTML=xmlDoc.getElementsByTagName("userinfo")[i].childNodes[1].firstChild.nodeValue;
                   
                   cell2.innerHTML=xmlDoc.getElementsByTagName("userinfo")[i].childNodes[0].firstChild.nodeValue;
                }

            };
            
            request.send();
        </script>
