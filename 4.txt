package com.services;

import javax.jws.WebService;
import javax.jws.WebMethod;
import javax.jws.WebParam;
import java.lang.Math;

/**
 *
 * @author student
 */
@WebService(serviceName = "Operations")
public class Operations {

    /**
     * Web service operation
     */
    @WebMethod(operationName = "evenOdd")
    public String evenOdd(@WebParam(name = "num") int num) {
        if (num % 2 == 0){
            return "Even";
        }
        return "Odd";
    }

    /**
     * Web service operation
     */
    @WebMethod(operationName = "factorial")
    public int factorial(@WebParam(name = "num") int num) {
        if (num ==0){
            return 1;
        }
        if (num < 0){
            return -1;
        }
        int fact = 1;
        for (;num > 0; num--){
            fact *= num;
        }
        return fact;
    }

    /**
     * Web service operation
     */
    @WebMethod(operationName = "isPrime")
    public boolean isPrime(@WebParam(name = "num") int num) {
        if (num <= 1){
            return false;
        }
        if (num == 2){
            return true;
        }
        for (int i = 3; i <= Math.sqrt(num); i++){
            if (num%i == 0){
                return false;
            }
        }
        return true;
    }

}

package com.services;

import javax.jws.WebService;
import javax.jws.WebMethod;
import javax.jws.WebParam;

/**
 *
 * @author student
 */
@WebService(serviceName = "EMI")
public class EMI {

    /**
     * Web service operation
     */
    @WebMethod(operationName = "EMI_reduced")
    public double EMI_reduced(@WebParam(name = "principal") long principal, @WebParam(name = "period") int period, @WebParam(name = "roi") double roi) {
        //TODO write your implementation code here:
        roi=roi/100;
        double emi=(principal+(principal*roi*period))/(period*12);
        return emi;
    }

    /**
     * Web service operation
     */
    @WebMethod(operationName = "EMI_Fixed")
    public double EMI_Fixed(@WebParam(name = "principal") long principal, @WebParam(name = "period") int period, @WebParam(name = "roi") double roi) {
        //TODO write your implementation code here:
        roi=roi/(12*100);
        double emi=(principal * roi * Math.pow(1 + roi, period)) / (Math.pow(1 + roi, period) - 1);
        return emi;
    }
}
