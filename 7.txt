Operations.svc:

using System;
using System.Collections.Generic;
using System.Linq;
using System.Runtime.Serialization;
using System.ServiceModel;
using System.Text;

namespace WcfService
{
    // NOTE: You can use the "Rename" command on the "Refactor" menu to change the class name "Operations" in code, svc and config file together.
    // NOTE: In order to launch WCF Test Client for testing this service, please select Operations.svc or Operations.svc.cs at the Solution Explorer and start debugging.
    public class Operations : IOperations
    {
        public string Grade(double marks)
        {
            string grd;
            if (marks >= 90)
            {
                grd = "O";
            }
            else if (marks >= 80)
            {
                grd = "A";
            }
            else if (marks >= 70)
            {
                grd = "B";
            }
            else if(marks >= 60)
            {
                grd = "C";
            }
            else if(marks >= 40)
            {
                grd = "D";
            }
            else
            {
                grd = "F";
            }
            return grd;
        }
    }
}

using System;

        protected void grade_Click(object sender, EventArgs e)
